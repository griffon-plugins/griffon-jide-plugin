16 package griffon.builder.swing.jide.factory
17 
18 import com.jidesoft.swing.Searchable
19 import griffon.builder.jide.factory.AbstractJideComponentFactory
20 import griffon.builder.jide.factory.DelegatingJideFactory
21 import griffon.plugins.jide.swing.support.SearchableBarWrapper
22 
23 /**
24  * @author Andres Almiray
25  */
26 class SearchableBarFactory extends AbstractJideComponentFactory implements DelegatingJideFactory {
27     public SearchableBarFactory() {
28         super(SearchableBarWrapper)
29     }
30 
31     public Object newInstance(FactoryBuilderSupport builder, Object name, Object value, Map properties) throws InstantiationException, IllegalAccessException {
32         FactoryBuilderSupport.checkValueIsNull(value, name)
33         def searchable = properties.remove("searchable")
34         if (!searchable) {
35             throw new IllegalArgumentException("Failed to create component for '$name ' reason: missing 'searchable' attribute")
36         }
37 
38         if (!(searchable instanceof Searchable)) {
39             throw new IllegalArgumentException("Failed to create component for '$name' reason: 'searchable' is not of type " + Searchable.class.name)
40         }
41 
42         String initialText = properties.remove('initialText') as String
43         boolean compact = properties.remove('compact') as boolean
44         if (initialText == null) initialText = ''
45         if (compact == null) compact = false
46         return new SearchableBarWrapper(searchable, initialText, compact)
47     }
48 
49     public boolean onHandleNodeAttributes(FactoryBuilderSupport builder, Object node,
50                                           Map attributes) {
51         setWidgetAttributes(builder, node.searchable, attributes, true)
52         return true
53     }
54 
55     public void setParent(FactoryBuilderSupport builder, Object parent, Object child) {
56         if (child.install) {
57             child.installOnContainer(parent)
58         }
59     }
60 }
