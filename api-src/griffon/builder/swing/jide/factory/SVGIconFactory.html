16 package griffon.builder.swing.jide.factory
17 
18 import com.kitfox.svg.SVGCache
19 import griffon.builder.jide.factory.AbstractJideComponentFactory
20 import griffon.plugins.jide.swing.support.ResizableSVGIcon
21 import org.codehaus.groovy.runtime.InvokerHelper
22 
23 import java.util.logging.Level
24 import java.util.logging.Logger
25 
26 /**
27  * @author Andres Almiray
28  */
29 class SVGIconFactory extends AbstractJideComponentFactory {
30     static Map aliases = [:]
31 
32     public SVGIconFactory() {
33         super(ResizableSVGIcon)
34     }
35 
36     public Object newInstance(FactoryBuilderSupport builder, Object name, Object value, Map properties) throws InstantiationException, IllegalAccessException {
37         FactoryBuilderSupport.checkValueIsNull(value, name)
38         String path = properties.remove("path")
39         ResizableSVGIcon svgIcon = new ResizableSVGIcon()
40         if (path) {
41             // handle svg aliases
42             String alias = path.substring(0, path.indexOf(":"))
43             if (alias && aliases.get(alias)) {
44                 path = aliases.get(alias) + (path - alias - ":")
45             }
46             InputStream is = null
47             try {
48                 is = Thread.currentThread().contextClassLoader.getResourceAsStream(path)
49                 URI uri = SVGCache.getSVGUniverse().loadSVG((InputStream) is, path)
50                 svgIcon.setSvgURI(uri)
51             } catch (Exception e) {
52                 Logger.getLogger("global").log(Level.WARNING, "Can't create SVG image from path '${path}'", e)
53             }
54         }
55 
56         return svgIcon
57     }
58 
59     public void setParent(FactoryBuilderSupport builder, Object parent, Object child) {
60         try {
61             MetaClass mc = InvokerHelper.getMetaClass(parent)
62             mc.setProperty(parent, "icon", child)
63             child.installSizeTracker(parent)
64             if (parent.preferredSizeSet && !child.sizeSet) {
65                 child.setSize(parent.preferredSize)
66             }
67         } catch (MissingPropertyException mpe) {
68             // ignore
69         }
70     }
71 
72     public static void registerSVGAlias(String alias, String path) {
73         aliases[alias] = path
74     }
75 }
